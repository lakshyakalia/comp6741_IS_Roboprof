PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://example.org/>
PREFIX dbpedia: <http://dbpedia.org/resource/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX vcard:      <http://www.w3.org/2001/vcard-rdf/3.0#>

### courses offered by the university. (Query 1)
SELECT DISTINCT ?courseName WHERE {
	dbpedia:Concordia_University ex:offersCourse ?object.
	?course ex:courseName ?courseName
}


### courses offered by the university with given subject (Query 4)
SELECT DISTINCT ?courseName WHERE {
	dbpedia:Concordia_University ex:offersCourse ?object.
	?course ex:courseSubject ?subject.
  	?course ex:courseName ?courseName
  FILTER(?subject="COMP"^^xsd:string)
}
