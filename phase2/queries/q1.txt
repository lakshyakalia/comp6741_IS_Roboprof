PREFIX ec: <http://eulergui.sourceforge.net/contacts.owl.n3#>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX ex: <http://example.org/>
PREFIX dbpedia: <http://dbpedia.org/resource/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX vcard:      <http://www.w3.org/2001/vcard-rdf/3.0#>

SELECT DISTINCT ?topic_label ?topic_uri WHERE{
	?course ex:courseSubject ?course_subject.
  	?course ex:courseNumber ?course_number.
  	?course ex:hasLecture ?lecture.
  	?lecture ex:hasTopicsCovered ?topic.
  	?topic rdfs:label ?topic_label.
  	?topic rdfs:seeAlso ?topic_uri
  FILTER(?course_subject='COMP' && ?course_number=6721)
}
