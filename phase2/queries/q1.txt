#P2 Query1
SELECT ?topics_label ?topics ?lecture_label ?lecture_content_label ?lecture_content WHERE{
	?course ex:courseName ?name.
  	?course ex:hasLecture ?lecture.
  	?lecture rdfs:label	?lecture_label.
  	?lecture ex:hasLectureContent ?lecture_content.
  	?lecture_content ex:hasTopicsCovered ?topics.
  	?lecture_content rdfs:label	?lecture_content_label.
  	?topics rdfs:label ?topics_label.
  FILTER(?name="INTELLIGENT SYSTEMS")
}
